# ============================================
# Workflow Orchestrator — System Requirements
# ============================================

# ── Runtime ──────────────────────────────────
PHP          >= 8.2       https://www.php.net/downloads
Composer     >= 2.0       https://getcomposer.org
Node.js      >= 18.0      https://nodejs.org
npm          >= 9.0       (comes with Node.js)

php-sqlite3     (default DB — can use MySQL instead)
php-pdo

php-mbstring
php-xml
php-curl
php-zip
php-bcmath
php-tokenizer
php-json
php-openssl
php-ctype

# ── Optional (for production) ─────────────────
Redis        >= 6.0       https://redis.io  (swap from DB queue driver)
MySQL        >= 8.0       https://mysql.com (swap from SQLite)
